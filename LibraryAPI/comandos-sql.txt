CREATE TABLE livro (
	id uuid not null primary key,
	isbn varchar(20) not null unique,
	titulo varchar(150) not null,
	data_publicacao date not null,
	genero varchar(30) not null,
	preco numeric(18, 2) not null,
	data_cadastro timestamp,
    data_atualizacao timestamp,
	data_ultima_atualizacao timestamp,
    id_usuario uuid,
	id_autor uuid not null references autor(id),
	constraint chk_genero check (
		genero in ('FICCAO', 'FANTASIA', 'MISTERIO', 'ROMANCE', 'BIOGRAFIA', 'CIENCIA',
		'AVENTURA', 'TERROR', 'RELIGIOSO', 'AUTOAJUDA')
	)
);

CREATE TABLE autor (
	id UUID PRIMARY KEY,
	nome VARCHAR(100) not null,
	data_nascimento date not null,
	nacionalidade VARCHAR(60) not null,
	data_cadastro timestamp not null,
	data_ultima_atualizacao timestamp not null
);

CREATE table users(
	id uuid primary key not null,
	login varchar(80) not null unique,
	senha varchar(255) not null,
	role varchar(20) not null,
	constraint chk_role check (role in ('GERENTE', 'OPERADOR'))

);
